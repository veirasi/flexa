function confirmarExclusao(cardId) { const card = document.getElementById(cardId); const container = card.parentElement; // Esconde o card da visão do lojista container.style.display = 'none'; // Cria o aviso (Toast) de "Desfazer" const toast = document.createElement('div'); toast.className = 'undo-toast'; toast.id = 'toast-desfazer'; toast.innerHTML = ` Cliente removido
Desfazer
`; document.body.appendChild(toast); // Se o lojista não clicar em nada em 4 segundos, o aviso some setTimeout(() => { const existingToast = document.getElementById('toast-desfazer'); if (existingToast) existingToast.remove(); }, 4000); } function desfazerExclusao(cardId) { const card = document.getElementById(cardId); const container = card.parentElement; // Traz o cliente de volta container.style.display = 'block'; card.style.transform = 'translateX(0)'; // Remove o aviso preto da tela imediatamente const toast = document.getElementById('toast-desfazer'); if (toast) toast.remove(); }
